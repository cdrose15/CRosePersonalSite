Factorial Calculation

Please enter a number: 

                        
                        // Function called during form submition    
                        function getFactorial() {
                            var num = document.getElementById("num").value
                            document.getElementById("fac").innerHTML = "Calculation = " + factorial(num);
                        }
                        
                        // Calculate factorial
                        function factorial(num) {
                    
                             // Factorial of 0 = 1
                             if (num == 0) {
                                return 1;
                             }
                             if (num < 0) {
                                return "Undefined - Please enter a positive number";
                             }
                           
                             // Any other number will take the user num * num-1 until num  = 1
                             else {
                                return num * factorial(num - 1);
                             }
                        }

Fizz Buzz

This Program Will Output Fizz / Buzz / FizzBizz Based on Multiples for 1-100

Please enter a number between 1 - 100:

Please enter a number between 1 - 100: 

                            
                            $(document).ready(function() {
                                $("#submit").click(function () {
                    
                                     // Getting values from each user inputted numbers
                                     var x = $("#a").val();
                                     var y = $("#b").val();

                                     // for loop for each instance of 1 - 100
                                     for (var i = 1; i <= 100; i++) {

                                         // If both numbers are multiples of each instance 1-100 print FizzBuzz
                                         if ((i % x == 0) && (i % y == 0)) {
                                            $("#result").append("FizzBuzz" + " ");
                                         }
                                          
                                         // If x is a multiple of each instance 1-100 print Fizz
                                         else if (i % x == 0) {
                                            $("#result").append("Fizz" + " ");
                                         }

                                         // If y is a multiple of each instance 1-100 print Buzz
                                         else if (i % y == 0) {
                                            $("#result").append("Buzz" + " ");
                                         }
                            
                                         // If neither number is a multiple it will print the number 1-100
                                         else {
                                            $("#result").append(i + " ");
                                         }
                                     }
                                 });
                            });

Is Your Word a Palindrome?

This Program Will Let You Know if Your String is a Palindrome

Enter a word: 

                            $(document).ready(function () {
                                 $("#palindrome").click(function () {
                                    var string = $("#word").val();

                                    // Convert String to Upper Case
                                    string = string.toUpperCase();

                                    // Remove White Space From Front and Back of String
                                    string = string.trim();

                                    // Split , Reverse, and Join to Reverse Entire String
                                    var reverseString = string.split('').reverse().join('');

                                    // If Statement to check if User Input = Revered User Input and Output
                                    if (string == reverseString) {
                                    $("#result").text("Yep! " + string + " is a Palindrome");
                                    }
                                    else {
                                    $("#result").text("Sorry! " + string + " is not a Palindrome");
                                    }
                                });
                             });
